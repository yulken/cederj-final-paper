\begin{lstlisting}[language=JavaScript, caption={A Entidade Game},captionpos=b, label=alg:entity]
    @Entity('games')
    export default class Game {
    @PrimaryGeneratedColumn('uuid')
    id: string;
    
    @Column()
    name: string;
    
    @Column()
    price: number;
    
    @Column()
    publisher: string;
    
    @Column()
    release_date: Date;
    
    @CreateDateColumn()
    created_at: Date;
    
    @UpdateDateColumn()
    updated_at: Date;
    }
\end{lstlisting}
