\begin{lstlisting}[language=JavaScript, caption={A Entidade Game.ts},captionpos=b, label=alg:game]
import {
    Entity,
    Column,
    PrimaryGeneratedColumn,
    CreateDateColumn,
    UpdateDateColumn,
    } from 'typeorm';
    
    @Entity('games')
    export default class Game {
    @PrimaryGeneratedColumn('uuid')
    id: string;
    
    @Column()
    name: string;
    
    @Column()
    price: number;
    
    @Column()
    publisher: string;
    
    @Column()
    release_date: Date;
    
    @CreateDateColumn()
    created_at: Date;
    
    @UpdateDateColumn()
    updated_at: Date;
    }
\end{lstlisting}
